1661.
select a.machine_id,  
round(
    (
        select avg(a1.timestamp) 
        from activity a1
        where a1.activity_type = 'end' and
        a1.machine_id=a.machine_id
    ) -
    (
        select avg(a1.timestamp)
        from activity a1
        where a1.activity_type='start' and
        a1.machine_id=a.machine_id
    ),3
) as processing_time 
from activity a
group by a.machine_id

solution 2:

    avg(case when activity_type='start' then timestamp end),3
) as processing_time
from activity
